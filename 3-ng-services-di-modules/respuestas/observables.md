  this.searchFormControl.valueChanges.pipe(
    filter(value=>!!value),
    debounceTime(300),
    distinctUntilChanged(),
    tap(()=>{this.isSearching = true;this.searchComplete = false;}),
  ).subscribe(value=>{
    console.log("value",value);
    this.isSearching = false;
    this.searchComplete = true;
  });
}
